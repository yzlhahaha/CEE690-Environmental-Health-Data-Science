co2_df=pd.read_csv('data_co2_weekly_mlo.txt',header=48,delim_whitespace=True)
# some quick_data_cleaning
# replace missing numbers with NaNs for clarity
missing=co2_df['ppm']<0
co2_df['ppm'].values[missing]=np.NaN
# initialize variables
times=co2_df['decimal']
co2=co2_df['ppm']
# missing values interpolation
not_observed=np.isnan(co2)
interpolator=interp1d(times[~not_observed],co2[~not_observed],kind='cubic')
co2_interpolated=interpolator(times)
co2_df['ppm_interpolated'] = co2_interpolated

normalizer=preprocessing.StandardScaler()
knn=neighbors.KNeighborsRegressor(n_neighbors=10)
pipe=pipeline.Pipeline([('norm',normalizer),('knn',knn)])
TSS=model_selection.TimeSeriesSplit(n_splits=10)

avg_mse = []
max_lag = 100
for n_lag in range(1, max_lag):
    n_max = len(co2_interpolated)
    X = np.zeros([n_max-n_lag, n_lag])
    for i in range(0, n_max - n_lag):
        X[i] = co2_interpolated[i: i+n_lag]
    y = co2_interpolated[n_lag:]
    scores=[]
    for train_ndx,valid_ndx in TSS.split(X,y):
        valid_ndx_adjusted=valid_ndx
        pipe.fit(X[train_ndx],y[train_ndx])
        score=(mean_squared_error\
              (y[valid_ndx_adjusted],pipe.predict(X[valid_ndx_adjusted])))
        scores.append(score)
    avg_mse.append(np.mean(scores))
    #print('With lag of {0}, average MSE of {1} with a standard error of {2}'.format(n_lag,np.mean(scores),np.std(scores)/np.sqrt(TSS.n_splits)))
avg_mse = np.array(avg_mse)

print('Starting iteration...')

    normalizer=preprocessing.StandardScaler()
    knn=neighbors.KNeighborsRegressor(n_neighbors=k)
    pipe=pipeline.Pipeline([('norm',normalizer),('knn',knn)])

    avg_mse = []
    for n_lag in range(1, max_lag):
        n_max = len(x)
        X = np.zeros([n_max-n_lag, n_lag])
        for i in range(0, n_max - n_lag):
            X[i] = x[i: i+n_lag]
        y = x[n_lag:]
        scores=[]
        for train_ndx,valid_ndx in TSS.split(X,y):
            valid_ndx_adjusted=valid_ndx
            pipe.fit(X[train_ndx],y[train_ndx])
            score=(mean_squared_error\
                  (y[valid_ndx_adjusted],pipe.predict(X[valid_ndx_adjusted])))
            scores.append(score)
        avg_mse.append(np.mean(scores))
        count += 1
        if (count % (total_count / 10)) == 0:
            print('In progress {}%'.format(count / total_count * 100))

    all_avg_mse.append(avg_mse)
print('Finished')
